	\begin{columns}

		\block{Requisites}{%
			\innerblock{Required packages}{%
				\begin{itemize}
					\item \textit{graphicx} to display images and logos
					\item \textit{tikz} to draw on slides
				\end{itemize}
			}
			\innerblock{Required documents}{%
				\begin{itemize}
					\item \textit{beamerthemehmgu.sty}, \textit{beamerouterthemehmgu.sty}, \textit{beamerinnerthemehmgu.sty} and \textit{beamercolorthemehmgu.sty} contain all the settings and commands
					\item \textit{hmgu\_color\_englisch.eps} image file for the Helmholtz Zentrum Logo
					\item \textit{image\_example.jpg} only to build the example presentation
				\end{itemize}
			}	
		}
		\column{0.6}
		\block{Available commands}{%
			\innerblock{About the poster}{%
				\begin{itemize}
					\item \texttt{$\backslash$title} title of the presentation; appears on the first slide and at the bottom of each slide (short version available)
					\item \texttt{$\backslash$subtitle} subtitle of the presentation; appears on the first slide
					\item \texttt{$\backslash$date} date of the presentation; appears on the first slide
				\end{itemize}
			}
			\innerblock{About the graphics}{%
				\begin{itemize}
					\item \texttt{$\backslash$slidelogo} logo to display on each slide on top right
					\item \texttt{$\backslash$titleimage} image to display on the first slide
				\end{itemize}	
			}
			\innerblock{About the authors}{%
				\begin{itemize}
					\item \texttt{$\backslash$author} name of the author; appears on the first slide and at the bottom of each slide (short version available)
					\item \texttt{$\backslash$email} email of the author; appears on the first slide
				\end{itemize}
			}
			\innerblock{About the institute}{%
				\begin{itemize}
					\item \texttt{$\backslash$institute} name of the institute; appears on the first slide, below the HMGU logo
				\end{itemize}
			}
		}
	\end{columns}
